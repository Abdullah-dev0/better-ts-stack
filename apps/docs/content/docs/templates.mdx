## Backend API Template (Express)

Selecting **Backend API** with Express generates:

    <File name=".env" />
    <File name=".env.example" />

    <File name="eslint.config.mjs" />
    <File name=".prettierrc" />
    <File name=".gitignore" />
  </Folder>
</Files>

<Callout type="info">
  The base backend template generates exactly two source files: `src/index.ts`
  and `src/routes/health.ts`. There is no `app.ts`, `routes/index.ts`,
  `middleware/errorHandler.ts`, `types/` directory, or `README.md` in the base
  output.
</Callout>

### src/index.ts

The entry point configures Express middleware, mounts the health route, and starts the server:

```typescript
import cors from "cors";
import dotenv from "dotenv";
import express, { Express, Request, Response } from "express";
import helmet from "helmet";
import morgan from "morgan";

import healthRouter from "./routes/health";

dotenv.config();

const app: Express = express();
const port = process.env.PORT || 3000;

app.use(helmet());
app.use(cors());
app.use(morgan("dev"));
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

app.use("/health", healthRouter);

// 404 handler
app.use((_req: Request, res: Response) => {
  res.status(404).json({ error: "Not Found" });
});

// Error handler
app.use((err: Error, _req: Request, res: Response) => {
  console.error(err.stack);
  res.status(500).json({ error: "Internal Server Error" });
});

app.listen(port, () => {
  console.log(`Server is running on http://localhost:${port}`);
});

export default app;
```

### Generated package.json scripts

```json
{
  "scripts": {
    "dev": "tsx watch src/index.ts",
    "build": "tsc",
    "start": "node dist/index.js",
    "lint": "eslint src",
    "lint:fix": "eslint src --fix",
    "format": "prettier --write \"src/**/*.ts\"",
    "type:check": "tsc --noEmit"
  }
}
```

> With **bun** as the package manager, `dev` becomes `bun --watch src/index.ts`.

## Full-stack Template (Next.js)

Selecting **Full-stack** generates a single Next.js 16 application at the project root. There are no separate `backend/` or `frontend/` subdirectories.

<Files>
  <Folder name="my-app" defaultOpen>
    <Folder name="app" defaultOpen>
      <File name="layout.tsx" />
      <File name="page.tsx" />
      <File name="globals.css" />
    </Folder>
    <Folder name="public">
      <File name="(static assets)" />
    </Folder>
    <File name="proxy.ts" />
    <File name="next.config.ts" />
    <File name="postcss.config.mjs" />
    <File name="tsconfig.json" />
    <File name="eslint.config.mjs" />
    <File name="next-env.d.ts" />
    <File name="package.json" />
    <File name=".env" />
    <File name=".env.example" />
    <File name=".gitignore" />
  </Folder>
</Files>

<Callout type="info">
  The Next.js base template does not generate `components/ui`, `lib/utils.ts`,
  or any example API client. Those can be added by you after generation.
</Callout>

### PostgreSQL + Prisma

  <Tab value="Next.js">
    <Files>
      <Folder name="prisma" defaultOpen>
        <File name="schema.prisma" />
      </Folder>
      <File name="prisma.config.ts" />

        <File name="prisma.ts" />

    Scripts added to `package.json`:
    ```bash
    npm run prisma:generate
    npm run prisma:migrate   # prisma migrate dev
    npm run prisma:studio
    ```

    Environment variable added: `DATABASE_URL`

    <Callout type="info">
      In Next.js projects, Prisma reads `DATABASE_URL` through `prisma.config.ts`.

    </Callout>

  </Tab>
</Tabs>

### PostgreSQL + Drizzle _(Next.js only)_

<Files>
  <Folder name="lib" defaultOpen>
    <File name="schema.ts" />
    <File name="db.ts" />
  </Folder>
  <File name="drizzle.config.ts" />
</Files>

Scripts added to `package.json`:

```bash
npm run db:generate  # drizzle-kit generate
npm run db:migrate   # drizzle-kit migrate
npm run db:studio    # drizzle-kit studio
```

Environment variable added: `DATABASE_URL`

<Callout type="tip">
  If Better Auth is also enabled with Drizzle, the generated `lib/schema.ts`
  contains the Better Auth core tables instead of a sample table.
</Callout>

## With Docker Module

<Files>
  <Folder name="project-root" defaultOpen>
    <File name="Dockerfile" />
    <File name="docker-compose.yml" />
    <File name=".dockerignore" />
  </Folder>
</Files>

The `Dockerfile` is framework-aware: it uses a Next.js standalone output stage for Full-stack projects and a Node.js multi-stage build for Express backends.

Scripts added to `package.json`:

```bash
npm run docker:build  # docker build -t <projectName> .
npm run docker:up     # docker-compose up -d
npm run docker:down   # docker-compose down
npm run docker:logs   # docker-compose logs -f
```

## Supported Combinations

| App type   | Database   | ORM/ODM  | Auth        | Status         |
| ---------- | ---------- | -------- | ----------- | -------------- |
| Backend    | none       | -        | JWT         | Yes          |
| Backend    | PostgreSQL | Prisma   | JWT         | Yes          |
| Backend    | MongoDB    | Mongoose | JWT         | Yes          |
| Full-stack | none       | -        | -           | Yes          |
| Full-stack | PostgreSQL | Prisma   | Better Auth | Yes          |
| Full-stack | PostgreSQL | Drizzle  | Better Auth | Yes          |
| Backend    | NestJS     | -        | -           | Coming soon |
